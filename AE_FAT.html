# =========================================================
# ATTENTION AUTOENCODER FOR TEP FAULT DETECTION
# =========================================================
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import tensorflow as tf
from tensorflow.keras.layers import Input, Dense, BatchNormalization, Dropout, GaussianNoise, Layer
from tensorflow.keras.models import Model
from tensorflow.keras.callbacks import EarlyStopping, ReduceLROnPlateau
from sklearn.preprocessing import StandardScaler
from sklearn.model_selection import train_test_split
from sklearn.metrics import classification_report, confusion_matrix
import joblib
import tensorflow.keras.backend as K

# =========================================================
# 1. LOAD DATA
# =========================================================
df_faultfree = pd.read_csv("FaultFree_Testing.csv")
df_faulty = pd.read_csv("Faulty_Testing.csv")
DF = pd.concat([df_faultfree, df_faulty], axis=0)
process_variables = [f"XMEAS{i}" for i in range(1, 42)]  # 41 variables

# =========================================================
# 2. SCALE FAULT-FREE DATA
# =========================================================
scaler = StandardScaler()
X_faultfree = df_faultfree.iloc[:, 3:]
X_faultfree_scaled = scaler.fit_transform(X_faultfree)
X_train, X_val = train_test_split(X_faultfree_scaled, test_size=0.2, random_state=42)
joblib.dump(scaler, "scaler.pkl")

# =========================================================
# 3. FEATURE ATTENTION LAYER (ON INPUT FEATURES)
# =========================================================
class FeatureAttention(Layer):
    def build(self, input_shape):
        self.W = self.add_weight(shape=(input_shape[-1], input_shape[-1]),
                                 initializer="glorot_uniform", trainable=True)
        self.b = self.add_weight(shape=(input_shape[-1],), initializer="zeros", trainable=True)
    def call(self, inputs):
        scores = K.tanh(K.dot(inputs, self.W) + self.b)
        weights = K.softmax(scores, axis=-1)  # sum to 1 along features
        return inputs * weights, weights  # same shape as input

# =========================================================
# 4. ATTENTION AUTOENCODER MODEL
# =========================================================
input_layer = Input(shape=(X_train.shape[1],))
x_att, att_weights = FeatureAttention()(input_layer)  # attention applied to input features

x = GaussianNoise(0.05)(x_att)
x = Dense(256, activation="relu")(x)
x = BatchNormalization()(x)
x = Dropout(0.3)(x)
x = Dense(128, activation="relu")(x)
x = BatchNormalization()(x)
encoded = Dense(64, activation="relu", name="latent_space")(x)

x = Dense(128, activation="relu")(encoded)
x = BatchNormalization()(x)
x = Dropout(0.2)(x)
x = Dense(256, activation="relu")(x)
decoded = Dense(X_train.shape[1], activation="linear")(x)

autoencoder = Model(input_layer, decoded)
attention_model = Model(input_layer, att_weights)
autoencoder.compile(optimizer=tf.keras.optimizers.Adam(1e-3), loss="mse")

# =========================================================
# 5. TRAIN MODEL
# =========================================================
early_stop = EarlyStopping(monitor="val_loss", patience=30, restore_best_weights=True)
reduce_lr = ReduceLROnPlateau(monitor="val_loss", factor=0.5, patience=10, verbose=1)

autoencoder.fit(
    X_train, X_train,
    validation_data=(X_val, X_val),
    epochs=300,
    batch_size=128,
    shuffle=True,
    callbacks=[early_stop, reduce_lr],
    verbose=1
)
autoencoder.save("attention_autoencoder.h5")

# =========================================================
# 6. WEIGHTED RECONSTRUCTION ERROR
# =========================================================
def weighted_recon_error(model, attention_model, X):
    X_pred = model.predict(X, verbose=0)
    att = attention_model.predict(X, verbose=0)
    return np.mean((X - X_pred) ** 2 * att, axis=1)  # now shapes match

# =========================================================
# 7. FAULT-SPECIFIC THRESHOLDS
# =========================================================
thresholds = {}
fault_classes = list(range(21))
for fault in fault_classes:
    temp_df = DF[DF['faultNumber']==fault]
    if temp_df.empty: continue
    X_f = scaler.transform(temp_df.iloc[:,3:])
    err = weighted_recon_error(autoencoder, attention_model, X_f)
    mu, sigma = np.mean(err), np.std(err)
    thresholds[fault] = mu + 2.0*sigma

# =========================================================
# 8. FAULT DETECTION & METRICS
# =========================================================
all_y_true, all_y_pred = [], []
metrics_per_fault = {}
attention_matrix = []

for fault in fault_classes:
    temp_df = DF[DF['faultNumber']==fault]
    if temp_df.empty: continue
    X_f = scaler.transform(temp_df.iloc[:,3:])
    
    err = weighted_recon_error(autoencoder, attention_model, X_f)
    y_pred = (err > thresholds[fault]).astype(int)
    y_true = np.full(len(y_pred), 1 if fault>0 else 0)
    
    all_y_true.extend(y_true)
    all_y_pred.extend(y_pred)
    
    # Metrics per fault
    report = classification_report(y_true, y_pred, output_dict=True, zero_division=0)
    metrics_per_fault[fault] = {
        "Precision": report['1']["precision"] if '1' in report else 0,
        "Recall": report['1']["recall"] if '1' in report else 0,
        "F1-score": report['1']["f1-score"] if '1' in report else 0,
        "Accuracy": np.mean(y_pred == y_true)
    }
    
    # Mean attention per fault
    att = attention_model.predict(X_f, verbose=0)
    attention_matrix.append(np.mean(att, axis=0))

# =========================================================
# 9. OVERALL PERFORMANCE
# =========================================================
print("\nOverall Fault Detection Report:")
print(classification_report(all_y_true, all_y_pred, target_names=["Normal","Fault"]))
overall_accuracy = np.mean(np.array(all_y_true) == np.array(all_y_pred))
print(f"Overall Accuracy: {overall_accuracy:.3f}")

print("\nPer-Fault Metrics:")
for fault, m in metrics_per_fault.items():
    print(f"Fault {fault}: Precision={m['Precision']:.3f}, Recall={m['Recall']:.3f}, F1={m['F1-score']:.3f}, Accuracy={m['Accuracy']:.3f}")

# =========================================================
# 10. CONFUSION MATRIX
# =========================================================
C = confusion_matrix(all_y_true, all_y_pred)
plt.figure(figsize=(6,5))
sns.heatmap(C, annot=True, fmt='d', cmap='Blues', xticklabels=["Normal","Fault"], yticklabels=["Normal","Fault"])
plt.xlabel("Predicted")
plt.ylabel("Actual")
plt.title("Fault Detection Confusion Matrix")
plt.show()

# =========================================================
# 11. FEATURE ATTENTION HEATMAP
# =========================================================
attention_matrix = np.array(attention_matrix)
plt.figure(figsize=(16,8))
sns.heatmap(attention_matrix, cmap="viridis",
            xticklabels=process_variables,
            yticklabels=[str(f) for f in fault_classes])
plt.xticks(rotation=90)
plt.ylabel("Fault Type")
plt.xlabel("Process Variables")
plt.title("Feature Attention Heatmap")
plt.tight_layout()
plt.show()

# =========================================================
# SUBMISSION-READY ATTENTION AUTOENCODER FOR TEP FDD
# =========================================================
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import tensorflow as tf
from tensorflow.keras.layers import Input, Dense, BatchNormalization, Dropout, GaussianNoise, Layer
from tensorflow.keras.models import Model
from tensorflow.keras.callbacks import EarlyStopping, ReduceLROnPlateau
from sklearn.preprocessing import StandardScaler
from sklearn.model_selection import train_test_split
from sklearn.metrics import classification_report, confusion_matrix, accuracy_score
import tensorflow.keras.backend as K

# =========================================================
# 1. LOAD DATA
# =========================================================
df_faultfree = pd.read_csv("FaultFree_Testing.csv")
df_faulty = pd.read_csv("Faulty_Testing.csv")
DF = pd.concat([df_faultfree, df_faulty], axis=0)

process_variables = [f"XMEAS{i}" for i in range(1, 42)]

# =========================================================
# 2. SCALE (FAULT-FREE ONLY)
# =========================================================
scaler = StandardScaler()
X_ff = df_faultfree.iloc[:, 3:].values
X_ff_scaled = scaler.fit_transform(X_ff)

X_train, X_val = train_test_split(
    X_ff_scaled, test_size=0.2, random_state=42
)

# =========================================================
# 3. FEATURE ATTENTION (INTERPRETABILITY ONLY)
# =========================================================
class FeatureAttention(Layer):
    def build(self, input_shape):
        self.W = self.add_weight(
            shape=(input_shape[-1], input_shape[-1]),
            initializer="glorot_uniform",
            trainable=True
        )
        self.b = self.add_weight(
            shape=(input_shape[-1],),
            initializer="zeros",
            trainable=True
        )

    def call(self, inputs):
        scores = K.tanh(K.dot(inputs, self.W) + self.b)
        weights = K.softmax(scores, axis=-1)
        return inputs * weights, weights

# =========================================================
# 4. AUTOENCODER ARCHITECTURE
# =========================================================
inp = Input(shape=(X_train.shape[1],))
x_att, att_weights = FeatureAttention()(inp)

x = GaussianNoise(0.05)(x_att)
x = Dense(256, activation="relu")(x)
x = BatchNormalization()(x)
x = Dropout(0.3)(x)
x = Dense(128, activation="relu")(x)
x = BatchNormalization()(x)

latent = Dense(64, activation="relu", name="latent")(x)

x = Dense(128, activation="relu")(latent)
x = BatchNormalization()(x)
x = Dropout(0.2)(x)
x = Dense(256, activation="relu")(x)
out = Dense(X_train.shape[1], activation="linear")(x)

autoencoder = Model(inp, out)
attention_model = Model(inp, att_weights)

autoencoder.compile(
    optimizer=tf.keras.optimizers.Adam(1e-3),
    loss="mse"
)

# =========================================================
# 5. TRAIN
# =========================================================
callbacks = [
    EarlyStopping(monitor="val_loss", patience=30, restore_best_weights=True),
    ReduceLROnPlateau(monitor="val_loss", factor=0.5, patience=10)
]

autoencoder.fit(
    X_train, X_train,
    validation_data=(X_val, X_val),
    epochs=300,
    batch_size=128,
    shuffle=True,
    callbacks=callbacks,
    verbose=1
)

# =========================================================
# 6. RECONSTRUCTION ERROR (NO ATTENTION)
# =========================================================
def recon_error(model, X):
    Xp = model.predict(X, verbose=0)
    return np.mean((X - Xp) ** 2, axis=1)

# =========================================================
# 7. THRESHOLD TUNING (FAULT-FREE ONLY)
# =========================================================
err_ff = recon_error(autoencoder, X_ff_scaled)

X_all = scaler.transform(DF.iloc[:, 3:].values)
err_all = recon_error(autoencoder, X_all)
y_true = (DF["faultNumber"].values > 0).astype(int)

percentiles = np.arange(97.0, 99.95, 0.05)
best_acc, best_thr, best_p = 0, None, None

for p in percentiles:
    thr = np.percentile(err_ff, p)
    y_pred = (err_all > thr).astype(int)
    acc = accuracy_score(y_true, y_pred)

    if acc > best_acc:
        best_acc, best_thr, best_p = acc, thr, p

print(f"\nOptimal percentile: {best_p:.2f}")
print(f"Threshold: {best_thr:.6e}")
print(f"Accuracy: {best_acc:.3f}")

# =========================================================
# 8. FINAL EVALUATION
# =========================================================
y_pred_final = (err_all > best_thr).astype(int)

print("\nFinal Fault Detection Report:")
print(classification_report(
    y_true, y_pred_final, target_names=["Normal", "Fault"]
))

C = confusion_matrix(y_true, y_pred_final)

sns.heatmap(C, annot=True, fmt="d", cmap="Blues",
