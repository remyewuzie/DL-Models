# 1. IMPORT LIBRARIES

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import tensorflow as tf
from tensorflow.keras.layers import (
    Input, Dense, BatchNormalization, Dropout, GaussianNoise, Layer
)
from tensorflow.keras.models import Model
from tensorflow.keras.callbacks import EarlyStopping
from sklearn.preprocessing import StandardScaler
from sklearn.model_selection import train_test_split
from sklearn.metrics import classification_report
import joblib
import tensorflow.keras.backend as K

# =========================================================
# 2. LOAD DATA
# =========================================================
df_faultfree = pd.read_csv("FaultFree_Testing.csv")
df_faulty = pd.read_csv("Faulty_Testing.csv")
DF = pd.concat([df_faultfree, df_faulty], axis=0)

# Process variables (XMEAS1XMEAS41)
process_variables = [
    "XMEAS1","XMEAS2","XMEAS3","XMEAS4","XMEAS5","XMEAS6","XMEAS7","XMEAS8","XMEAS9",
    "XMEAS10","XMEAS11","XMEAS12","XMEAS13","XMEAS14","XMEAS15","XMEAS16","XMEAS17",
    "XMEAS18","XMEAS19","XMEAS20","XMEAS21",
    "XMEAS22","XMEAS23","XMEAS24","XMEAS25","XMEAS26","XMEAS27","XMEAS28","XMEAS29",
    "XMEAS30","XMEAS31","XMEAS32","XMEAS33","XMEAS34","XMEAS35","XMEAS36","XMEAS37",
    "XMEAS38","XMEAS39","XMEAS40","XMEAS41"
]

# =========================================================
# 3. SCALE FAULT-FREE DATA (TRAINING ONLY)
# =========================================================
scaler = StandardScaler()
X_faultfree = df_faultfree.iloc[:, 3:]
X_faultfree_scaled = scaler.fit_transform(X_faultfree)

X_train, X_val = train_test_split(
    X_faultfree_scaled, test_size=0.2, random_state=42
)

joblib.dump(scaler, "scaler.pkl")

# =========================================================
# 4. FEATURE ATTENTION LAYER
# =========================================================
class FeatureAttention(Layer):
    def build(self, input_shape):
        self.W = self.add_weight(
            shape=(input_shape[-1], input_shape[-1]),
            initializer="glorot_uniform",
            trainable=True
        )
        self.b = self.add_weight(
            shape=(input_shape[-1],),
            initializer="zeros",
            trainable=True
        )

    def call(self, inputs):
        scores = K.tanh(K.dot(inputs, self.W) + self.b)
        weights = K.softmax(scores, axis=1)
        return inputs * weights, weights

# =========================================================
# 5. ATTENTION-BASED AUTOENCODER
# =========================================================
input_layer = Input(shape=(X_train.shape[1],))

x = GaussianNoise(0.05)(input_layer)
x = Dense(128, activation="relu")(x)
x = BatchNormalization()(x)
x = Dropout(0.3)(x)

x = Dense(64, activation="relu")(x)
x = BatchNormalization()(x)

x_att, att_weights = FeatureAttention()(x)

encoded = Dense(32, activation="relu", name="latent_space")(x_att)

x = Dense(64, activation="relu")(encoded)
x = Dense(128, activation="relu")(x)
decoded = Dense(X_train.shape[1], activation="linear")(x)

autoencoder = Model(input_layer, decoded)
attention_model = Model(input_layer, att_weights)

autoencoder.compile(
    optimizer=tf.keras.optimizers.Adam(learning_rate=1e-3),
    loss="mse"
)

# =========================================================
# 6. TRAIN MODEL
# =========================================================
early_stop = EarlyStopping(
    monitor="val_loss", patience=30, restore_best_weights=True
)

autoencoder.fit(
    X_train, X_train,
    epochs=200,
    batch_size=256,
    shuffle=True,
    validation_data=(X_val, X_val),
    callbacks=[early_stop],
    verbose=1
)

autoencoder.save("attention_autoencoder.h5")

# =========================================================
# 7. RECONSTRUCTION ERROR & THRESHOLD
# =========================================================
def reconstruction_error(model, X):
    X_hat = model.predict(X, verbose=0)
    return np.mean((X - X_hat) ** 2, axis=1)

X_ff_all = scaler.transform(df_faultfree.iloc[:, 3:])
err_ff = reconstruction_error(autoencoder, X_ff_all)

mu, sigma = np.mean(err_ff), np.std(err_ff)
threshold = mu + 2.5 * sigma

# =========================================================
# 8. FAULT DETECTION EVALUATION (BINARY)
# =========================================================
y_true, y_pred = [], []

for fault in range(21):
    temp = DF[DF["faultNumber"] == fault]
    X_temp = scaler.transform(temp.iloc[:, 3:])
    err = reconstruction_error(autoencoder, X_temp)

    pred = (err > threshold).astype(int)
    true = np.full(len(pred), 1 if fault > 0 else 0)

    y_true.extend(true)
    y_pred.extend(pred)

print("\nFault Detection Performance:")
print(classification_report(y_true, y_pred, target_names=["Normal", "Fault"]))

# =========================================================
# 9. ATTENTION HEATMAP (INTERPRETABILITY)
# =========================================================
# Normal condition
att_normal = attention_model.predict(X_ff_all, verbose=0)
mean_att_normal = np.mean(att_normal, axis=0)

# Example faulty condition (Fault 5)
X_fault5 = scaler.transform(
    DF[DF["faultNumber"] == 5].iloc[:, 3:]
)
att_fault5 = attention_model.predict(X_fault5, verbose=0)
mean_att_fault5 = np.mean(att_fault5, axis=0)

heatmap_data = np.vstack([mean_att_normal, mean_att_fault5])

plt.figure(figsize=(14, 4))
sns.heatmap(
    heatmap_data,
    cmap="viridis",
    xticklabels=process_variables,
    yticklabels=["Normal", "Fault 5"]
)
plt.xticks(rotation=90)
plt.ylabel("Operating Condition")
plt.title("Feature Attention Heatmap")
plt.tight_layout()
plt.show()

from sklearn.metrics import classification_report, confusion_matrix

# =========================================================
# FAULT DETECTION & PER-FAULT METRICS
# =========================================================
fault_classes = list(range(21))  # Normal (0) + Faults 1-20
all_y_true, all_y_pred = [], []
metrics_per_fault = {}

for fault in fault_classes:
    temp_df = DF[DF['faultNumber'] == fault]
    X_faulty = scaler.transform(temp_df.iloc[:, 3:])
    faulty_reconstruction_loss = reconstruction_error(autoencoder, X_faulty)

    # Use adaptive threshold per fault if needed, or global threshold
    y_pred = (faulty_reconstruction_loss > threshold).astype(int)
    y_true = np.full(len(y_pred), 1 if fault > 0 else 0)

    all_y_true.extend(y_true)
    all_y_pred.extend(y_pred)

    # Compute metrics per fault (binary)
    report = classification_report(y_true, y_pred, output_dict=True, zero_division=0)
    metrics_per_fault[fault] = {
        "Precision": report['1']["precision"] if '1' in report else 0,
        "Recall": report['1']["recall"] if '1' in report else 0,
        "F1-score": report['1']["f1-score"] if '1' in report else 0,
        "Accuracy": np.mean(y_pred == y_true)
    }

# =========================================================
# OVERALL REPORT
# =========================================================
print("\nOverall Fault Detection Report (Binary):")
print(classification_report(all_y_true, all_y_pred, target_names=["Normal", "Fault"]))
overall_accuracy = np.mean(np.array(all_y_true) == np.array(all_y_pred))
print(f"Overall Accuracy: {overall_accuracy:.3f}")

# =========================================================
# CONFUSION MATRIX
# =========================================================
C = confusion_matrix(all_y_true, all_y_pred)
plt.figure(figsize=(6, 5))
sns.heatmap(C, annot=True, fmt='d', cmap='Blues', xticklabels=["Normal","Fault"], yticklabels=["Normal","Fault"])
plt.xlabel("Predicted")
plt.ylabel("Actual")
plt.title("Fault Detection Confusion Matrix")
