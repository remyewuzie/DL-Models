# Import necessary libraries
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import tensorflow as tf
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import LSTM, Dense, Dropout, BatchNormalization
from tensorflow.keras.utils import to_categorical
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import confusion_matrix, classification_report, roc_auc_score, f1_score, log_loss, recall_score

# Load Data
train = pd.read_csv('FaultFree_Training.csv')
test = pd.read_csv('FaultFree_Testing.csv')

# Sort data by 'simulationRun' and 'faultNumber'
train.sort_values(['simulationRun', 'faultNumber'], inplace=True)
test.sort_values(['simulationRun', 'faultNumber'], inplace=True)

# Splitting data into features and labels
X_train = train.drop(columns=['faultNumber', 'simulationRun']).values  # Convert to NumPy
y_train = train['faultNumber'].values  # Target labels

X_test = test.drop(columns=['faultNumber', 'simulationRun']).values
y_test = test['faultNumber'].values

# Standardize features for better LSTM training
scaler = StandardScaler()
X_train = scaler.fit_transform(X_train)
X_test = scaler.transform(X_test)

# Reshape input for LSTM (samples, time steps, features)
X_train = X_train.reshape(X_train.shape[0], 1, X_train.shape[1])
X_test = X_test.reshape(X_test.shape[0], 1, X_test.shape[1])

# Fix categorical encoding issue by shifting labels to start from 0
y_train = y_train - y_train.min()  
y_test = y_test - y_test.min()

# Get the correct number of classes
num_classes = len(np.unique(y_train))  

# Convert labels to categorical (One-Hot Encoding)
y_train = to_categorical(y_train, num_classes=num_classes)
y_test = to_categorical(y_test, num_classes=num_classes)

# Build Robust LSTM Model
model = Sequential([
    LSTM(128, input_shape=(X_train.shape[1], X_train.shape[2]), return_sequences=True, activation='sigmoid'),
    BatchNormalization(),
    LSTM(128, return_sequences=False, activation='sigmoid'),
    Dropout(0.4),
    Dense(300, activation='relu'),
    Dropout(0.4),
    Dense(num_classes, activation='softmax')
])

# Compile Model
model.compile(loss='categorical_crossentropy', optimizer='adam', metrics=['accuracy'])

# Print Model Summary
print(model.summary())

# Train Model
history = model.fit(X_train, y_train, epochs=100, batch_size=256, validation_data=(X_test, y_test), verbose=2)

# Plot Training History
plt.figure(figsize=(10, 5))
plt.plot(history.history['accuracy'], label='Training Accuracy', color='b')
plt.plot(history.history['val_accuracy'], label='Validation Accuracy', color='r')
plt.xlabel("Epochs")
plt.ylabel("Accuracy")
plt.legend()
plt.grid()
plt.title("Training & Validation Accuracy")
plt.show()

# Evaluate Model
score, acc = model.evaluate(X_test, y_test, verbose=0)
print(f"\nTest Accuracy: {acc:.3f}")
print(f"Test Loss: {score:.3f}")

# Predictions
y_pred_prob = model.predict(X_test)
y_pred = np.argmax(y_pred_prob, axis=1)
y_true = np.argmax(y_test, axis=1)

# Classification Report
print("\nClassification Report:")
print(classification_report(y_true, y_pred))

# Compute Confusion Matrix
labels = np.unique(y_true)
C = confusion_matrix(y_true, y_pred, labels=labels)

# Compute Class-Wise Accuracy
class_wise_accuracy = C.diagonal() / C.sum(axis=1)

# Print Accuracy for Each Fault
print("\nClass-Wise Accuracy (Per Fault):")
for label, acc in zip(labels, class_wise_accuracy):
    print(f"Fault {label}: {acc:.3f}")

# Compute Overall Accuracy
overall_accuracy = C.diagonal().sum() / C.sum()
print(f"\nOverall Average Accuracy: {overall_accuracy:.3f}")

# Compute Fault Detection Rate (FDR)
TP = C.diagonal()
Total_Fault_Cases = C.sum(axis=1)
FDR = np.where(Total_Fault_Cases != 0, TP / Total_Fault_Cases, 0)

# Print FDR for each fault
print("\nFault Detection Rate (FDR) for Each Fault:")
for label, fdr in zip(labels, FDR):
    print(f"Fault {label}: {fdr:.3f}")

# Compute Overall FDR
overall_FDR = np.sum(TP) / np.sum(Total_Fault_Cases)
print(f"\nOverall FDR: {overall_FDR:.3f}")

# Define Heatmap Color Scheme
cmap = sns.light_palette("green", as_cmap=True)

# Plot Class-Wise Accuracy
plt.figure(figsize=(12, 5))
plt.bar(labels, class_wise_accuracy, color='skyblue')
plt.xlabel("Fault Number")
plt.ylabel("Accuracy")
plt.title("Class-Wise Accuracy for Each Fault")
plt.xticks(labels, rotation=90)
plt.ylim(0, 1)
plt.show()

# Compute Recall & Precision Matrices
A = C / C.sum(axis=1, keepdims=True)  # Recall Matrix (Row-normalized)
B = C / C.sum(axis=0, keepdims=True)  # Precision Matrix (Column-normalized)

# Confusion Matrix Heatmap
plt.figure(figsize=(15, 10))
sns.heatmap(C, annot=True, cmap=cmap, fmt=".3f", xticklabels=labels, yticklabels=labels)
plt.xlabel('Predicted Class')
plt.ylabel('Actual Class')
plt.title("Confusion Matrix")
plt.show()

# Precision Matrix Heatmap
plt.figure(figsize=(15, 10))
sns.heatmap(B, annot=True, cmap=cmap, fmt=".3f", xticklabels=labels, yticklabels=labels)
plt.xlabel('Predicted Class')
plt.ylabel('Actual Class')
plt.title("Precision Matrix")
plt.show()

# Recall Matrix Heatmap
plt.figure(figsize=(15, 10))
sns.heatmap(A, annot=True, cmap=cmap, fmt=".3f", xticklabels=labels, yticklabels=labels)
plt.xlabel('Predicted Class')
plt.ylabel('Actual Class')
plt.title("Recall Matrix")
plt.show()
