# ============================================================
# CNN + Attention + Fault-wise & Temporal Attention Analysis
# (SINGLE CELL  FINAL, EXPERIMENT-READY)
# ============================================================

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import tensorflow as tf
from tensorflow.keras.layers import Input, Conv1D, MaxPooling1D, Dense, Dropout, Layer
from tensorflow.keras.models import Model
from tensorflow.keras.optimizers import Adam
from tensorflow.keras.callbacks import EarlyStopping
from tensorflow.keras import backend as K
from sklearn.preprocessing import StandardScaler
from sklearn.model_selection import train_test_split
from sklearn.metrics import classification_report, accuracy_score
from sklearn.utils.class_weight import compute_class_weight
from scipy.stats import entropy

# -------------------------
# 1. Load Data
# -------------------------
df_faultfree = pd.read_csv("FaultFree_Testing.csv")
df_faulty = pd.read_csv("Faulty_Testing.csv")

df_faultfree["faultNumber"] = 0
df_faulty = df_faulty.dropna(subset=["faultNumber"])
df_faulty["faultNumber"] = df_faulty["faultNumber"].astype(int)

df = pd.concat([df_faultfree, df_faulty], ignore_index=True)

X = df.drop(columns=["faultNumber"]).values
y = df["faultNumber"].values

# -------------------------
# 2. Train-Test Split
# -------------------------
X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.2, stratify=y, random_state=42
)

# -------------------------
# 3. Normalize
# -------------------------
scaler = StandardScaler()
X_train = scaler.fit_transform(X_train)
X_test = scaler.transform(X_test)

X_train = X_train[..., np.newaxis]
X_test = X_test[..., np.newaxis]

n_classes = len(np.unique(y_train))

# -------------------------
# 4. Attention Layer
# -------------------------
class Attention1D(Layer):
    def build(self, input_shape):
        self.W = self.add_weight(shape=(input_shape[-1],1),
                                 initializer="glorot_uniform",
                                 trainable=True)
        self.b = self.add_weight(shape=(input_shape[1],1),
                                 initializer="zeros",
                                 trainable=True)

    def call(self, x):
        e = K.tanh(K.dot(x, self.W) + self.b)
        a = K.softmax(e, axis=1)
        x = x * a
        return K.sum(x, axis=1)

# -------------------------
# 5. CNN + Attention Model
# -------------------------
inputs = Input(shape=(X_train.shape[1], 1))
x = Conv1D(64, 3, activation="relu", padding="same")(inputs)
x = MaxPooling1D(2, padding="same")(x)
x = Conv1D(128, 3, activation="relu", padding="same")(x)
x = MaxPooling1D(2, padding="same")(x)

attention_out = Attention1D()(x)
x = Dense(64, activation="relu")(attention_out)
x = Dropout(0.4)(x)
outputs = Dense(n_classes, activation="softmax")(x)

model = Model(inputs, outputs)
model.compile(
    optimizer=Adam(1e-4),
    loss="sparse_categorical_crossentropy",
    metrics=["accuracy"]
)

# -------------------------
# 6. Class Weights
# -------------------------
class_weights = compute_class_weight(
    class_weight="balanced",
    classes=np.unique(y_train),
    y=y_train
)
class_weights = dict(enumerate(class_weights))

# -------------------------
# 7. Train
# -------------------------
early_stop = EarlyStopping(monitor="val_loss", patience=10, restore_best_weights=True)

model.fit(
    X_train, y_train,
    validation_data=(X_test, y_test),
    epochs=100,
    batch_size=256,
    class_weight=class_weights,
    callbacks=[early_stop],
    verbose=1
)

# -------------------------
# 8. Evaluation
# -------------------------
y_pred = np.argmax(model.predict(X_test), axis=1)

print(f"\nOverall Accuracy: {accuracy_score(y_test, y_pred)*100:.2f}%\n")
print("Per-Class Accuracy:")
for f in np.unique(y_test):
    idx = np.where(y_test == f)[0]
    print(f"Fault {f}: {accuracy_score(y_test[idx], y_pred[idx])*100:.2f}%")

print("\nClassification Report:")
print(classification_report(y_test, y_pred))

# -------------------------
# 9. Extract Attention Weights
# -------------------------
class AttentionExplain(Layer):
    def build(self, input_shape):
        self.W = self.add_weight(shape=(input_shape[-1],1),
                                 initializer="glorot_uniform",
                                 trainable=True)
        self.b = self.add_weight(shape=(input_shape[1],1),
                                 initializer="zeros",
                                 trainable=True)

    def call(self, x):
        e = K.tanh(K.dot(x, self.W) + self.b)
        return K.softmax(e, axis=1)

conv_output = model.layers[4].output
attention_weights = AttentionExplain()(conv_output)
attention_model = Model(model.input, attention_weights)

A = attention_model.predict(X_test).squeeze(-1)

# -------------------------
# 10. TEP Variable Names
# -------------------------
tep_variables = [f"XMEAS{i}" for i in range(1, 42)]

# -------------------------
# 11. Fault-wise Attention Heatmap
# -------------------------
unique_faults = np.unique(y_test)
avg_attention = np.zeros((len(unique_faults), A.shape[1]))

for i, f in enumerate(unique_faults):
    avg_attention[i] = A[y_test == f].mean(axis=0)

mapped_attention = np.zeros((len(unique_faults), len(tep_variables)))
for i in range(len(unique_faults)):
    mapped_attention[i] = np.interp(
        np.linspace(0, avg_attention.shape[1]-1, len(tep_variables)),
        np.arange(avg_attention.shape[1]),
        avg_attention[i]
    )

plt.figure(figsize=(18,6))
plt.imshow(mapped_attention, aspect="auto", cmap="viridis")
plt.colorbar(label="Attention Weight")
plt.yticks(range(len(unique_faults)), [f"Fault {f}" for f in unique_faults])
plt.xticks(range(len(tep_variables)), tep_variables, rotation=90)
plt.title("Fault-wise Attention Heatmap (TEP Variables)")
plt.tight_layout()
plt.show()

# -------------------------
# 12. TEMPORAL STABILITY OF ATTENTION
# -------------------------
temporal_stages = {
    "Early": (0.0, 0.3),
    "Middle": (0.3, 0.7),
    "Late": (0.7, 1.0)
}

temporal_attention = {}

for fault in unique_faults:
    idx = np.where(y_test == fault)[0]
    A_fault = A[idx]
    n = len(A_fault)

    temporal_attention[fault] = {}
    for stage, (l, u) in temporal_stages.items():
        temporal_attention[fault][stage] = A_fault[int(l*n):int(u*n)].mean(axis=0)

temporal_mapped = {}
for fault in unique_faults:
    temporal_mapped[fault] = {}
    for stage in temporal_stages:
        temporal_mapped[fault][stage] = np.interp(
            np.linspace(0, temporal_attention[fault][stage].shape[0]-1, len(tep_variables)),
            np.arange(temporal_attention[fault][stage].shape[0]),
            temporal_attention[fault][stage]
        )

# -------------------------
# 13. Temporal Heatmap (Representative Fault)
# -------------------------
fault_to_plot = unique_faults[1]

temporal_stack = np.vstack([
    temporal_mapped[fault_to_plot]["Early"],
    temporal_mapped[fault_to_plot]["Middle"],
    temporal_mapped[fault_to_plot]["Late"]
])

plt.figure(figsize=(18,4))
plt.imshow(temporal_stack, aspect="auto", cmap="viridis")
plt.colorbar(label="Attention Weight")
plt.yticks([0,1,2], ["Early", "Middle", "Late"])
plt.xticks(range(len(tep_variables)), tep_variables, rotation=90)
plt.title(f"Temporal Evolution of Attention  Fault {fault_to_plot}")
plt.tight_layout()
plt.show()

# -------------------------
# 14. Quantitative Attention Shift (KL Divergence)
# -------------------------
e = temporal_mapped[fault_to_plot]["Early"]
m = temporal_mapped[fault_to_plot]["Middle"]
l = temporal_mapped[fault_to_plot]["Late"]

e, m, l = e/e.sum(), m/m.sum(), l/l.sum()

print("\nKL Divergence:")
print(f"Early  Middle: {entropy(e, m):.4f}")
print(f"Middle  Late: {entropy(m, l):.4f}")
